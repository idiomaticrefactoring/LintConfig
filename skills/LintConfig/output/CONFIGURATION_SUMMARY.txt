   cp CheckstyleConfig_long.xml /path/to/your/project/

   For Maven (pom.xml):
   <plugin>
       <groupId>org.apache.maven.plugins</groupId>
       <artifactId>maven-checkstyle-plugin</artifactId>
       <configuration>
           <configLocation>CheckstyleConfig_long.xml</configLocation>
           <includeTestSourceDirectory>true</includeTestSourceDirectory>
       </configuration>

   checkstyle {
